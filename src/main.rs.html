 9 fn main() {
10     loop {
11         let result = readline("Kiss! ");
12         match result {
13             Some(input)
14                 => {
15                     add_history(input.as_slice());
16                     for token in tokenize(input) {
17                         match token {
18                             Ok(token)
19                                 => {
20                                     println!("      {:?}", token) }
21                             Err(error)
22                                 => {
23                                     println!("      {}", error);
24                                     break },}}}
25             None
26                 => {
27                     break }}}}

29 fn tokenize(input: String) -> Vec<Result<Token, String>> {
30     input
31         .split(CharExt::is_whitespace)
32         .map(assign_token)
33         .collect() }
